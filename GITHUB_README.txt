//Zur Benutzung eines vorhandenen GitHubs
0. git clone https://github.com/PieterBu/CPS.git
1. git pull   					-- Zum Aktualisieren der Ordner
2. git add .  					-- Hinzufügen neuer Dateien
3. git commit -m "Nachricht" 			-- Erkennt die Änderungen
4. git push					-- Hochladen der Änderungen

//Erstellen eines neuen GitHubs
1. git init					-- Im entsprechendem Ordner ausführen
2. git remote add origin https://github.com/USERNAME/OTHERREPOSITORY.git	--Verlinken
3. git pull origin master 			-- Zum Aktualisieren der Ordner
4. git add .  					-- Hinzufügen neuer Dateien
5. git commit -m "Nachricht" 			-- Erkennt die Änderungen
6. git push --set-upstream origin master	-- Hochladen der Änderungen
